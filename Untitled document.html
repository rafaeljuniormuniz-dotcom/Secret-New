<script>

// PASSWORD (plain)

const ADMIN_PASSWORD = 'secret';

// Toast

function showToast(msg,time=2500){const t=document.getElementById('toast');t.textContent=msg;t.style.display='block';clearTimeout(t._t);t._t=setTimeout(()=>t.style.display='none',time);}

// Messages

function getMessages(){try{return JSON.parse(localStorage.getItem('sf_messages')||'[]');}catch(e){return[]}}

function saveMessages(arr){localStorage.setItem('sf_messages',JSON.stringify(arr));}

// Form

const form=document.getElementById('applyForm');

form.addEventListener('submit',e=>{

e.preventDefault();

const name=document.getElementById('name').value.trim()||'(anônimo)';

const why=document.getElementById('why').value.trim();

const phone=document.getElementById('phone').value.trim();

if(!why||!phone){showToast('Preencha todos os campos obrigatórios');return}

const msgs=getMessages();

msgs.unshift({id:Date.now(),name,phone,why,date:new Date().toISOString(),read:false});

saveMessages(msgs);

form.reset();

showToast('Candidatura enviada — obrigado.');

});

// Admin

const adminArea=document.getElementById('adminArea');

const msgsTableBody=document.querySelector('#msgsTable tbody');

function renderAdmin(){const msgs=getMessages();msgsTableBody.innerHTML='';if(msgs.length===0){msgsTableBody.innerHTML='<tr><td colspan="5" style="opacity:0.6">Nenhuma mensagem encontrada.</td></tr>';return}

for(const m of msgs){

const tr=document.createElement('tr');

const tdName=document.createElement('td');tdName.textContent=m.name;

const tdPhone=document.createElement('td');tdPhone.textContent=m.phone;

const tdWhy=document.createElement('td');tdWhy.textContent=m.why;

const tdDate=document.createElement('td');tdDate.textContent=new Date(m.date).toLocaleString();

const tdActions=document.createElement('td');

const btnDel=document.createElement('button');btnDel.textContent='Apagar';btnDel.className='btn btn-ghost';btnDel.style.padding='6px 8px';

btnDel.addEventListener('click',()=>{if(!confirm('Apagar esta mensagem?'))return;const arr=getMessages().filter(x=>x.id!==m.id);saveMessages(arr);renderAdmin();});

const btnMark=document.createElement('button');btnMark.textContent=m.read?'Marcado':'Marcar lida';btnMark.className='btn';btnMark.style.padding='6px 8px';btnMark.style.background=m.read?'linear-gradient(90deg,#bfbfbf,#a69a3a)':'linear-gradient(90deg,var(--gold),#c99c2f)';

btnMark.addEventListener('click',()=>{const arr=getMessages();for(const it of arr)if(it.id===m.id)it.read=true;saveMessages(arr);renderAdmin();});

tdActions.appendChild(btnMark);tdActions.appendChild(btnDel);

tr.appendChild(tdName);tr.appendChild(tdPhone);tr.appendChild(tdWhy);tr.appendChild(tdDate);tr.appendChild(tdActions);

msgsTableBody.appendChild(tr);

}

}

// Login

function promptAdminLogin(){const pw=prompt('Senha do Painel:');if(!pw)return false;if(pw===ADMIN_PASSWORD){adminArea.style.display='block';document.querySelector('main').style.display='none';renderAdmin();showToast('Acesso concedido');return true}else{alert('Senha incorreta');return false}}

document.getElementById('btn-open-admin').addEventListener('click',()=>{promptAdminLogin();});

document.getElementById('btn-admin').addEventListener('click',()=>{promptAdminLogin();});

document.getElementById('btn-logout').addEventListener('click',()=>{if(confirm('Sair do painel?')){adminArea.style.display='none';document.querySelector('main').style.display='grid';showToast('Logout realizado');}});

document.getElementById('btn-clear').addEventListener('click',()=>{if(!confirm('Apagar todas as mensagens?'))return;localStorage.removeItem('sf_messages');renderAdmin();showToast('Mensagens apagadas');});

document.getElementById('btn-export').addEventListener('click',()=>{const arr=getMessages();if(arr.length===0){showToast('Sem mensagens para exportar');return}const csv=[['Nome','Telefone','Mensagem','Data']];for(const r of arr)csv.push([r.name.replace(/"/g,'""'),r.phone,r.why.replace(/"/g,'""'),new Date(r.date).toLocaleString()]);const csvContent=csv.map(r=>r.map(c=>`"${c}"`).join(',')).join('\n');const blob=new Blob([csvContent],{type:'text/csv;charset=utf-8;'});const url=URL.createObjectURL(blob);const a=document.createElement('a');a.href=url;a.download='candidaturas_secret_central.csv';a.click();URL.revokeObjectURL(url);});

// Phone placeholder

document.getElementById('org-phone').textContent = '+55 19 9XXXX-XXXX';

// Stars animation

const starsContainer = document.getElementById('stars');

for(let i=0;i<60;i++){

const s=document.createElement('div');s.className='star';

s.style.left=Math.random()*100+'%';

s.style.top=Math.random()*100+'%';

s.style.width=1+Math.random()*2+'px';

s.style.height=s.style.width;

s.style.animationDuration=2+Math.random()*3+'s';

starsContainer.appendChild(s);

}
